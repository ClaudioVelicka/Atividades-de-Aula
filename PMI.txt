Sub Ex_05()
    
    
Dim a As Double
Dim b As Double
Dim c As Double
Dim x2 As Double
Dim x1 As Double

a = InputBox("Informe o valor de A: ")
b = InputBox("Informe o valor de B: ")
c = InputBox("Informe o valor de C: ")

x1 = (-b + Sqr(b * b - 4 * a * c)) / (2 * a)
x2 = (-b - Sqr(b * b - 4 * a * c)) / (2 * a)

MsgBox ("Os valores das raízes são " & x1 & " e " & x2)
    

End Sub

Sub Ex_16()

Dim H As Double
Dim valorH As Double
Dim descendentes As Double
Dim salaliq As Double
Dim porcentual As Double

H = InputBox("Informe a quantidade de horas trabalhadas: ")
valorH = InputBox("Informe o valor por hora: ")
descendentes = InputBox("Informe o número de descendentes: ")

salaliq = (H * valorH) - (H * valorH * porcentual) + (100 * descendentes)

MsgBox ("O salária será: " & salaliq)

End Sub
